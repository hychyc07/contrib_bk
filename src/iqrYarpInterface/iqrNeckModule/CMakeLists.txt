SET(PROJECTNAME moduleYarpNeck)
PROJECT(${PROJECTNAME})

ADD_DEFINITIONS(-DQT_THREAD_SUPPORT -D_REENTRANT -DQT_NO_DEBUG -DIQRMODULE)

SET(QT_MT_REQUIRED TRUE)
FIND_PACKAGE(Qt3 REQUIRED)
FIND_PACKAGE(YARP REQUIRED)
FIND_PACKAGE(YARPIFLIB REQUIRED)

SET(folder_header moduleYarpNeck.hpp)
SET(folder_source moduleYarpNeck.cpp)

INCLUDE_DIRECTORIES(/usr/include/iqr ${QT_INCLUDE_DIR} ${YARPIFLIB_INCLUDE_DIRS})

ADD_LIBRARY(${PROJECTNAME} SHARED ${folder_header} ${folder_source})
TARGET_LINK_LIBRARIES(${PROJECTNAME} ${YARPIFLIB_LIBRARIES})

SET_TARGET_PROPERTIES(${PROJECTNAME} PROPERTIES PREFIX "")
