SET(PROJECTNAME moduleIcubAttention)
PROJECT(${PROJECTNAME})

ADD_DEFINITIONS(-DQT_THREAD_SUPPORT -D_REENTRANT -DQT_NO_DEBUG -DIQRMODULE)

SET(QT_MT_REQUIRED TRUE)
FIND_PACKAGE(Qt4 REQUIRED)
FIND_PACKAGE(YARP REQUIRED)

SET(folder_header moduleIcubAttention.hpp)
SET(folder_source moduleIcubAttention.cpp)

INCLUDE_DIRECTORIES(/usr/include/iqr ${QT_INCLUDE_DIR})

ADD_LIBRARY(${PROJECTNAME} SHARED ${folder_header} ${folder_source})

TARGET_LINK_LIBRARIES(${PROJECTNAME} ${YARP_LIBRARIES})

SET_TARGET_PROPERTIES(${PROJECTNAME} PROPERTIES PREFIX "")

INSTALL(TARGETS ${PROJECTNAME} LIBRARY DESTINATION $ENV{HOME}/iqr/lib/Modules )
