CMAKE_MINIMUM_REQUIRED(VERSION 2.6)

SET(KEYWORD "zdfControl")

PROJECT(${KEYWORD})

FIND_PACKAGE(ICUB)
FIND_PACKAGE(YARP REQUIRED)
SET (CMAKE_MODULE_PATH ${CMAKE_MODULE_PATH} ${ICUB_DIR}/conf ${YARP_DIR}/conf)
FIND_PACKAGE(GtkMM REQUIRED)
FIND_PACKAGE(GtkPlus REQUIRED)

FIND_PACKAGE(Gthread REQUIRED)
# set up to use GTK+
ADD_DEFINITIONS(${GTKPLUS_C_FLAGS})
LINK_LIBRARIES(${GtkPlus_LIBRARIES})
LINK_DIRECTORIES(${GTKPLUS_LIBRARY_DIR})
INCLUDE_DIRECTORIES(${GtkPlus_INCLUDE_DIRS})

#set up gthread
ADD_DEFINITIONS(${GTHREAD_C_FLAGS})
LINK_LIBRARIES(${Gthread_LIBRARIES})
INCLUDE_DIRECTORIES(${Gthread_INCLUDE_DIRS})

#search for includes
INCLUDE_DIRECTORIES(
    ${PROJECT_SOURCE_DIR}/include
    ${ICUB_INCLUDE_DIRS}
    ${YARP_INCLUDE_DIRS}
    ${GtkPlus_INCLUDE_DIRS}
)

# Search for source code.
FILE(GLOB folder_source *.cpp *.cc .c)
FILE(GLOB folder_header *.h)
SOURCE_GROUP("Source Files" FILES ${folder_source})
SOURCE_GROUP("Header Files" FILES ${folder_header})


# Set up our main executable.
ADD_EXECUTABLE(${KEYWORD} ${folder_source} ${folder_header} )
TARGET_LINK_LIBRARIES( ${KEYWORD} ${YARP_LIBRARIES} ${GTKPLUS_LINK_FLAGS} ${GtkPlus_LIBRARIES} )	

install(TARGETS ${KEYWORD} DESTINATION bin)
