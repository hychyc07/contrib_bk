
cmake_minimum_required(VERSION 2.6)

SET(PROJECTNAME sendBlobs2Eye2World)
PROJECT(${PROJECTNAME})

FIND_PACKAGE(YARP REQUIRED)
FIND_PACKAGE(ICUB)

set(CMAKE_MODULE_PATH ${YARP_MODULE_PATH} ${CMAKE_MODULE_PATH})
set(CMAKE_MODULE_PATH ${ICUB_MODULE_PATH} ${CMAKE_MODULE_PATH})



FIND_PACKAGE(OpenCV REQUIRED)

SET(CMAKE_MODULE_PATH ${CMAKE_MODULE_PATH} ${ICUB_MODULE_PATH} ${OpenCV_MODULE_PATH})

ADD_DEFINITIONS(${OpenCV_DEFINITIONS} )


INCLUDE_DIRECTORIES(${OPENCV_INCLUDE_DIR} ${ICUB_INCLUDE_DIRS} ${YARP_INCLUDE_DIRS})	

SET(folder_source cvaux.cpp   cvcolor.cpp    cvlabel.cpp  sendBlobs.cpp cvblob.cpp  cvcontour.cpp  cvtrack.cpp) 
SET(folder_header cvblob.h)

SOURCE_GROUP("Source Files" FILES ${folder_source} )

ADD_EXECUTABLE(${PROJECTNAME} ${folder_source})

TARGET_LINK_LIBRARIES(${PROJECTNAME} ${OPENCV_LIBRARIES} ${YARP_LIBRARIES})

INSTALL(TARGETS ${PROJECTNAME} DESTINATION $ENV{ICUB_ROOT}/bin)
