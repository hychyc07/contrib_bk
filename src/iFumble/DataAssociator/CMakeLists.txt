cmake_minimum_required(VERSION 2.6)

set(PROJECTNAME dataAssoc)
PROJECT(${PROJECTNAME})

FIND_PACKAGE(YARP REQUIRED)
FIND_PACKAGE(ICUB REQUIRED)

set(CMAKE_MODULE_PATH ${YARP_MODULE_PATH} ${CMAKE_MODULE_PATH})
set(CMAKE_MODULE_PATH ${ICUB_MODULE_PATH} ${CMAKE_MODULE_PATH})


FIND_PACKAGE(OpenCV REQUIRED)
FIND_PACKAGE(Qt3 REQUIRED)

SET(CMAKE_MODULE_PATH ${CMAKE_MODULE_PATH} ${ICUB_MODULE_PATH} ${OpenCV_MODULE_PATH})

ADD_DEFINITIONS(${QT_DEFINITIONS} )
ADD_DEFINITIONS(${OpenCV_DEFINITIONS} )
ADD_DEFINITIONS(-DQT_DLL)
ADD_DEFINITIONS(-DQT_THREAD_SUPPORT)
ADD_DEFINITIONS(-Wall -O2)

INCLUDE_DIRECTORIES(${ICUB_INCLUDE_DIRS} ${YARP_INCLUDE_DIRS} ${QT_INCLUDE_DIR} ${OPENCV_INCLUDE_DIR} ${QT_INCLUDE_PATH} ${CMAKE_CURRENT_BINARY_DIR} ${QT_QTSQL_INCLUDE_DIR} "include")
 

file(GLOB folder_header include/*.h)
file(GLOB folder_source src/*.cpp)


SOURCE_GROUP("Source Files" FILES ${folder_source})


ADD_EXECUTABLE(${PROJECTNAME} ${folder_source})
TARGET_LINK_LIBRARIES(${PROJECTNAME} ${OPENCV_LIBRARIES} ${YARP_LIBRARIES} ${QT_LIBRARIES} ${ICUB_LIBRARIES}) 
INSTALL(TARGETS ${PROJECTNAME} DESTINATION $ENV{ICUB_ROOT}/bin)
